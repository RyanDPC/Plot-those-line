% Header configuration for Pandoc → PDF (LaTeX)
% Uses fancyhdr to place a banner image in the header on all pages
\usepackage{graphicx}
\usepackage{fancyhdr}
\usepackage{lastpage}
\usepackage{geometry}

% Example: 2.5cm margins; tweak if necessary
\geometry{margin=2.5cm}

\setlength{\headheight}{48pt}

% Fancy header/footer setup

\renewcommand{\headrulewidth}{0pt}

% Left header: banner image. Path is relative to repository root when running pandoc there.

% Footer: page X / Y (optional — remove if not required)
